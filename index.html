# 快速开始

在 main.js 中引入组件库

// 全部引入
import 'jx-ui/dist/css/index.css'
import JUI from "jx-ui";
Vue.use(JUI);

// 按需引入
import 'jx-ui/dist/css/demo.css'
import { Demo } from "jx-ui";
Vue.use(Demo);
